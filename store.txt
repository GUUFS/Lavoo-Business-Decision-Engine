// paypal.tsx
import { useEffect, useRef, useState } from "react";

interface PayPalCheckoutProps {
  amount: number;
  onSuccess: (result: any) => void;
  onError?: (error: any) => void;
}

declare global {
  interface Window {
    paypal?: {
      Buttons: (options: any) => { 
        render: (element: HTMLElement | null) => Promise<void>;
        close?: () => void;
      };
    };
  }
}

export default function PayPalCheckout({ amount, onSuccess, onError }: PayPalCheckoutProps) {
  const paypalRef = useRef<HTMLDivElement | null>(null);
  const buttonsRef = useRef<any>(null);
  const [sdkReady, setSdkReady] = useState(false);
  const [isRendering, setIsRendering] = useState(false);
  const mountedRef = useRef(true);

  // Load PayPal SDK only once
  useEffect(() => {
    const loadPayPalScript = async () => {
      console.log("PayPal client:", import.meta.env.VITE_PAYPAL_CLIENT_ID);

      // Check if script already exists
      let script = document.querySelector(`script[src*="paypal.com/sdk/js"]`) as HTMLScriptElement | null;

      if (!script) {
        script = document.createElement("script");
        script.src = `https://www.paypal.com/sdk/js?client-id=${import.meta.env.VITE_PAYPAL_CLIENT_ID}&currency=USD`;
        script.async = true;
        
        script.onload = () => {
          console.log("PayPal SDK loaded successfully");
          setSdkReady(true);
        };

        script.onerror = () => {
          console.error("Failed to load PayPal SDK");
          onError?.("Failed to load PayPal");
        };

        document.body.appendChild(script);
      } else {
        // Script exists, check if PayPal is ready
        if (window.paypal) {
          console.log("PayPal SDK already loaded");
          setSdkReady(true);
        } else {
          // Wait for SDK to be ready
          let attempts = 0;
          const checkPayPal = setInterval(() => {
            if (window.paypal) {
              console.log("PayPal SDK ready after waiting");
              clearInterval(checkPayPal);
              setSdkReady(true);
            } else if (attempts++ > 20) {
              clearInterval(checkPayPal);
              console.error("PayPal SDK timeout");
              onError?.("PayPal initialization timeout");
            }
          }, 150);
        }
      }
    };

    loadPayPalScript();

    return () => {
      mountedRef.current = false;
    };
  }, []); // Run only once on mount

  // Render buttons when SDK is ready
  useEffect(() => {
    if (!sdkReady || !window.paypal || !paypalRef.current || isRendering) {
      return;
    }

    const renderButtons = async () => {
      if (!mountedRef.current || !paypalRef.current) return;

      setIsRendering(true);

      try {
        // Clear any existing buttons
        if (paypalRef.current) {
          paypalRef.current.innerHTML = '';
        }

        // Clear previous button reference
        if (buttonsRef.current) {
          buttonsRef.current = null;
        }

        console.log("Creating PayPal buttons for amount:", amount);

        const buttons = window.paypal!.Buttons({
          style: {
            layout: 'vertical',
            color: 'blue',
            shape: 'rect',
            label: 'paypal'
          },

          createOrder: async () => {
            try {
              console.log("Creating order for amount:", amount);
              const res = await fetch("http://localhost:8000/api/paypal/create-order", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ amount: amount.toString() }),
              });

              const text = await res.text();
              console.log("Create order response:", text);

              if (!res.ok) {
                throw new Error(`Failed to create order: ${text}`);
              }

              if (!text) {
                throw new Error("Empty response from create-order");
              }

              const data = JSON.parse(text);
              
              if (!data.id) {
                throw new Error(`Invalid order response: ${text}`);
              }

              console.log("Order created successfully:", data.id);
              return data.id;
            } catch (err) {
              console.error("Create order error:", err);
              onError?.(err);
              throw err;
            }
          },

          onApprove: async (data: { orderID: string }) => {
            try {
              console.log("Capturing order:", data.orderID);
              const res = await fetch("http://localhost:8000/api/paypal/capture-order", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ order_id: data.orderID }),
              });

              const text = await res.text();
              console.log("Capture order response:", text);

              if (!res.ok) {
                throw new Error(`Capture failed: ${text}`);
              }

              if (!text) {
                throw new Error("Empty response from capture-order");
              }

              const result = JSON.parse(text);
              console.log("Payment captured successfully:", result);
              onSuccess(result);
            } catch (err) {
              console.error("Capture order error:", err);
              onError?.(err);
            }
          },

          onCancel: (data: any) => {
            console.log("Payment cancelled by user:", data);
          },

          onError: (err: any) => {
            console.error("PayPal Button Error:", err);
            onError?.(err);
          }
        });

        if (buttons && paypalRef.current && mountedRef.current) {
          await buttons.render(paypalRef.current);
          buttonsRef.current = buttons;
          console.log("PayPal buttons rendered successfully");
        }
      } catch (err) {
        console.error("Error rendering PayPal buttons:", err);
        onError?.(err);
      } finally {
        setIsRendering(false);
      }
    };

    renderButtons();

    // Cleanup function
    return () => {
      if (paypalRef.current) {
        paypalRef.current.innerHTML = '';
      }
      buttonsRef.current = null;
    };
  }, [sdkReady, amount]); // Re-render when amount changes

  if (!sdkReady) {
    return (
      <div className="text-center py-6">
        <div className="animate-spin w-8 h-8 border-3 border-blue-600 border-t-transparent rounded-full mx-auto mb-2"></div>
        <p className="text-sm text-gray-600">Loading PayPal...</p>
      </div>
    );
  }

  return (
    <div className="w-full">
      <div ref={paypalRef} className="w-full min-h-[150px]"></div>
      {isRendering && (
        <div className="text-center text-sm text-gray-500 mt-2">
          Initializing payment...
        </div>
      )}
    </div>
  );
}
